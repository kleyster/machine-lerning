def closure_graph(synset,fn):
    seen = set()
    node = set()
    edge = list()  
    def recurse(s):
        if not s in seen:
            seen.add(s)
#             print(s.lemma_names('eng')[0])
            node.add(s.lemma_names('eng')[0])
            for s1 in fn(s):
                node.add(s1.lemma_names('eng')[0])
                edge.append((s.lemma_names('eng')[0],s1.lemma_names('eng')[0]))
                recurse(s1)
        return (node,edge)
    return recurse(synset)
